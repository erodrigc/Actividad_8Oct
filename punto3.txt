-- Se crea vista "Accesories_Low_Units"  para listar aquellos accesorios de los cuales se tienen cantidades por debajo de 5
CREATE OR REPLACE VIEW "Accesories_Low_Units" AS
SELECT A."id", A."type" Name, A."code", B."name" Manufacturer
FROM "Options" A
INNER JOIN "Manufacturers" B ON A."manufacturer_id" = B."id"
WHERE A."units_available" < 5
;

SELECT * FROM "Accesories_Low_Units"
